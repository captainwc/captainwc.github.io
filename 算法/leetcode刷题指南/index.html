int maxProfit(int k, vector<int>& prices) {
    vector<int> buy(k+1, -prices[0]);
    vector<int> sell(k+1, 0);
    buy[0] = 0;
    for(int i = 1; i < prices.size(); ++i) {
        for(int j = 1; j <= k; ++j) {
            buy[j] = max(buy[j], sell[j - 1] - prices[i]);
            sell[j] = max(sell[j], buy[j] + prices[i]);
        }
    }
    return sell[k];
}
